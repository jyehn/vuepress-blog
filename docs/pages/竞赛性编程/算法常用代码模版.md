##  快速排序

1. yxc模板

``` cpp
void quick_sort(int q[], int l, int r) {
    if (l >= r) return;
    
    int i = l - 1, j = r + 1, x = q[l + r >> 1];
    while (i < j) {
        while(q[++ i] < x);
        while(q[-- j] > x);
        if (i < j)
            swap(q[i], q[j]);
    }
    quick_sort(q, l, j);
    quick_sort(q, j+1, r);
}
```

2. 其他模板

``` cpp

void partition(int A[], int left, int right) {
   int pivot = A[left];
    while (left < right) {
        while (left < right && A[right] > pivot)
            ++right;
        A[left] = A[right];
        while (left < right && A[left] <= pivot)
            --left;
        A[right] = A[left];
    }
    A[left] = pivot;
    return left;
}
void quick_sort(int A[], int left, int right) {
    if (left < right) {
        int pos = partition(A, left, right);
        quick_sort(A, left, pos - 1);
        quick_sort(A, pos + 1, right);
    }
}

```

## 快速选择（求第k大的数）

```cpp
int quick_select(int q[], int l, int r, int k)
{
    if (l >= r) return q[l];

    int i = l - 1, j = r + 1, x = q[l + r >> 1];
    while (i < j)
    {
        do i ++ ; while (q[i] < x);
        do j -- ; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }

    if (j - l + 1 >= k) return quick_select(q, l, j, k);
    else return quick_select(q, j + 1, r, k - (j - l + 1));
}

```

## 整数二分

整数二分的本质是，在整个区间中，存在一个明显的分界线，分界线的两边对于某一个性质，一边满足，一边不满足。

`check`的取法和区间的向哪边缩短是有对应关系的

如果要找的是右半边的左端点，那么check函数应该设置成右半边满足的性质，对应每次`check`为真的时候，二分的边界往左边缩小，`r = mid`，这种情况下不需要改动`mid`

如果要找的是左半边的右端点，那么check函数应该设置成左半边满足的性质，对应每次`check`为真的时候，二分的边界往右边缩小, `l = mid`，而这种情况下可能会有l在每次循环后都是同一个数，陷入死循环，在不改动循环跳出条件的情况下，我们设置`mid = (l + r + 1) / 2`

```cpp
bool check(int x) {/* ...  */} 

/*
区间[l, r]被划分成[l, mid]和[mid + 1, r]时使用
*/
int right_interval_min(int l, int r) {
    while (l < r) {
        int mid = l + r >> 1;
        if (check(mid)) r = mid;
        else l = mid + 1;
    }
    return l;
}
/*
区间[l, r]被划分成[l, mid - 1]和[mid, r]时使用
*/
int left_interval_max(int l, int r) {
    while (l < r) {
        int mid = l + r  + 1>> 1;
        if (check(mid)) l = mid;
        else r = mid - 1;
    }
    return l;
}
```

```cpp
struct BigInteger {
    static const int BASE = 100000000;            //压8位
    static const int WIDTH = 8;
    vector<int> s;                                //从低位到高位存储
    
    BigInteger(long long num = 0) { *this = num;} //构造函数
    BigInteger operator = (long long num) {         //赋值运算符
        s.clear();
       	do {
            s.push_back(num % BASE);
            num /= BASE;
        } while (num > 0);
        return *this;
    }
    BigInteger operator = (const string& str) {
        s.clear();
        int x, len = (str.length() - 1) / WIDTH + 1;
        for (int i = 0; i < len; ++i) {
            int end = str.length() - i * WIDTH;    //从低位读入 确定开始和结束
            int start = max(0, end - WIDTH);
            sscanf(str.substr(start, end - start).c_str(), "%d", &x);
            s.push_back(x);
        }
        return *this;
    }
    ostream& operator << (ostream &out, const BigInteger& x) {
        out << x.s.back();
        for (int i = x.s.size() - 2; i >= 0; --i) {
            char buff[20];
            sprintf(buf, "%08d", x.s[i]);
            for (int j = 0; j < strlen(buf); ++j) out << buf[j];
        }
        return out;
    }
    istream& operator >> (istream &in, BigInteger& x) {
        string s;
        if (!(in >> s)) return in;
        x = s;
        return in;
    }
    BigInteger operator + (const BigInteger& b) const {
        BigInteger c;
        c.s.clear();
        for (int i = 0, g = 0; ; ++i) {
            if (g == 0 && i >= s.size() && i >= b.size()) break;
            int x = g;
            if (i < s.size()) x += s[i];
            if (i < b.s.size()) x += b.s[i];
            c.s.push_back(x % BASE);
            g = x / BASE;
        }
        return c;
    }
}
```

## 一维前缀和和一维差分

```
假设有一个数组a[1], a[2], ..., a[n]
那么定义前缀和为:
s[i] = a[1] + a[2] + ... + a[i]
求数组a在[l, r]区间内的和就可以转换为:
a[l] + ... + a[r] = s[r] - s[l - 1]


定义差分为：
d[i] = a[i] - a[i - 1]
那么对数组a区间[l, r]每个数加上c，等价于
d[l] += c, d[r + 1] -= c;
再将数组d求前缀和，即可得到加上c之后的数组a

一个数组差分再求前缀和就是它本身，但是这个操作可以将对区间的加法变成对端点的加减

```

## 二维前缀和和二维差分

```
s[i, j] = 第i行第j列格子左上部分所有元素的和
以(x1, y1)为左上角，(x2, y2)为右下角的子矩阵的和为：
s[x2, y2] - s[x1 - 1, y2] - s[x2, y1 - 1] + s[x1 - 1][y1 - 1]
二维差分的构造：
假定初始矩阵为0， 再遍历原矩阵，每个元素执行一次插入
给以(x1, y1)为左上角，(x2, y2)为右下角的子矩阵中的所有元素加上c：
S[x1, y1] += c, S[x2 + 1, y1] -= c, S[x1, y2 + 1] -= c, S[x2 + 1, y2 + 1] += c（可以依据容斥原理推导）
```

## Trie树

``` cpp
int son[N][26], cnt[N], idx;
```

``` cpp
    int i = low, j = i * 2;
    while (j <= high) {
        if (j + 1 <= high && heap[j + 1] > heap[j])
```

## Kruskal算法

``` cpp
    }

        a = father[a];

        father[x] = a;
        z = x;
    }

        father[i] = i;
```

##  lc模板(带rank)

``` cpp
class UnionFindSet {
public:
    UnionFindSet(int n) {
        ranks_ = vector<int>(n + 1, 0);        
        parents_ = vector<int>(n + 1, 0);                
        
        for (int i = 0; i < parents_.size(); ++i)
            parents_[i] = i;
    }
    
    // Merge sets that contains u and v.
    // Return true if merged, false if u and v are already in one set.
    bool Union(int u, int v) {
        int pu = Find(u);
        int pv = Find(v);
        if (pu == pv) return false;
        
        // Meger low rank tree into high rank tree
        if (ranks_[pv] < ranks_[pu])
            parents_[pv] = pu;           
        else if (ranks_[pu] < ranks_[pv])
            parents_[pu] = pv;
        else {
            parents_[pv] = pu;
            ranks_[pu] += 1;
        }
        
        return true;
    }
    
    // Get the root of u.
    int Find(int u) {        
        // Compress the path during traversal
        if (u != parents_[u])
            parents_[u] = Find(parents_[u]);        
        return parents_[u];
    }
private:
    vector<int> parents_;
    vector<int> ranks_;
};
```
